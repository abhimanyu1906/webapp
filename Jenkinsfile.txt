pipeline {
	agent any
	
	stages{
		stage('Build'){
            		steps {
                		sh 'mvn clean package'
            		}
            		post {
                		success {
                    			archiveArtifacts artifacts: '**/target/*.war'
                		}
            		}
		}
		stage('deploy') {
			steps {
				sh 'scp -i /var/lib/jenkins/workspace/ab/DEV1.pem ./target/*.war ec2-user@172.31.30.172:/root/apache-tomcat-7.0.53/webapps'
			}
		}
	}
}
